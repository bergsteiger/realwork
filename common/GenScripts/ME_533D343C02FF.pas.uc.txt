 begin
  with Result do
  begin
   Right := aRight + cWindowCaptionButtonPadding;
   Top := aTop + cWindowCaptionButtonPadding;
   Bottom := Top + l_ButtonHeight - cBugFixButtonSizeDecrease;
   Left := Right + l_ButtonWidth - cBugFixButtonSizeDecrease;

 end;//lp_MakeButtonRect
